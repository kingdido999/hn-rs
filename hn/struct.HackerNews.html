Main interface to the Hacker News API

Examples

extern crate hn;
use std::time::Duration;
use std::thread;

fn main() {
  let hn = hn::HackerNews::new();
  while hn.into_iter().count() == 0 {
      thread::sleep(Duration::from_millis(1000));
  }
  for item in hn.into_iter() {
      println!("item: {}", item.title());
  }
}
